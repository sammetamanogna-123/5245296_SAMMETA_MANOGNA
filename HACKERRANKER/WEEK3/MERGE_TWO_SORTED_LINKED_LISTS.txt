#include<stdio.h>

SinglyLinkedListNode* mergeLists(SinglyLinkedListNode* head1, SinglyLinkedListNode* head2) {
    SinglyLinkedListNode dummy;
    dummy.next = NULL;
    SinglyLinkedListNode* tail = &dummy;
    while (head1 != NULL && head2 != NULL) {
        if (head1->data < head2->data) {
            tail->next = head1;
            head1 = head1->next;
        }
        else {
            tail->next = head2;
            head2 = head2->next;
        }
        tail =tail->next;
    }
    if (head1 != NULL) 
        tail->next = head1;
    if (head2 != NULL) 
        tail->next = head2;
    return dummy.next;

}
