#include<stdio.h>

char* isBalanced(char* s) {
    int n = strlen(s);
    char stack[n];
    int top = -1;
    for(int i=0;i<n;i++){
        char ch=s[i];
        if(ch=='(' || ch=='{' || ch=='[') {
            stack[++top] = ch;
        }
        else {
            if(top == -1)
                return "NO";
            char open = stack[top--];
            if((ch == ')' && open != '(') || (ch == '}' && open !='{') || (ch == ']' && open !='[')) {
                return "NO";
            }
        }
    }
    return (top == -1) ? "YES" : "NO";
    

}
