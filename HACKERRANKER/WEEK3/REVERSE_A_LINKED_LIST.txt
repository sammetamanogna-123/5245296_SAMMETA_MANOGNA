#include<stdio.h>

SinglyLinkedListNode* reverse(SinglyLinkedListNode* llist) {
    SinglyLinkedListNode *prev = NULL;
    SinglyLinkedListNode *cur = llist;
    SinglyLinkedListNode *next = NULL;
    while (cur != NULL) {
        next = cur->next;
        cur->next = prev;
        prev = cur;
        cur = next;
    }
    return prev;

}
