#include<stdio.h>

int cmpfunc(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}
int maxMin(int k, int arr_count, int* arr) {
    qsort(arr, arr_count, sizeof(int), cmpfunc);
    int min_unfair = arr[arr_count - 1];
    for (int i = 0; i <= arr_count - k; i++) {
        int diff = arr[i + k - 1] - arr[i];
        if (diff < min_unfair){
            min_unfair = diff;
        }
    }
    return min_unfair;
}
