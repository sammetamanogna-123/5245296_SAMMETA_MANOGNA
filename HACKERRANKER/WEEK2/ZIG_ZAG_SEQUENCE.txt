#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int cmpfunc(const void* a, const void* b) {
    return (*(int*)a - *(int*)b);
}
void findZigZagSequence(int a[], int n) {
    int temp;
    qsort(a, n, sizeof(int), cmpfunc);
    int mid = (n + 1) / 2 - 1;
    temp = a[mid];
    a[mid] = a[n - 1];
    a[n - 1] = temp;
    int st = mid + 1;
    int ed = n - 2;
    while (st <= ed) {
        temp = a[st];
        a[st] = a[ed];
        a[ed] = temp;
        st++;
        ed--;
    }
    for (int i = 0; i < n; i++) {
        printf("%d", a[i]);
        if (i != n - 1) printf(" ");
    }
    printf("\n");
}
int main() {

    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    int t;
    scanf("%d", &t);
    while(t--) {
        int n;
        scanf("%d", &n);
        int a[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &a[i]);
        }
        findZigZagSequence(a, n);
    }
        
    return 0;
}
