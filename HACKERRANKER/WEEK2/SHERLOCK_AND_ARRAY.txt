#include<stdio.h> 

char* balancedSums(int arr_count, int* arr) {
    int t = 0;
    for (int i = 0; i < arr_count; i++) {
        t += arr[i];
    }
    int l = 0;
    for (int i = 0; i < arr_count; i++) {
        t -= arr[i];
        if (l == t) {
            return "YES";
        }
        l += arr[i];
    }
    return "NO";

}
