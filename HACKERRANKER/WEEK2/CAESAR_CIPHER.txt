#include<stdio.h>

char* caesarCipher(char* s, int k) {
    int n = strlen(s);
    char* result = (char*)malloc((n + 1) * sizeof(char));
    k = k % 26;
    for (int i = 0; i < n; i++) {
        char ch = s[i];
        if (ch >= 'a' && ch <= 'z') {
            result[i] = ((ch - 'a' + k) % 26) + 'a';
        }
        else if (ch >= 'A' && ch <= 'Z') {
            result[i] = ((ch - 'A' + k) % 26) + 'A';
        }
        else {
            result[i] = ch;
        }
    }
    result[n] = '\0';
    return result;

}
