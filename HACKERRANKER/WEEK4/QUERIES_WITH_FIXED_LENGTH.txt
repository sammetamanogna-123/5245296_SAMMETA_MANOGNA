#include<stdio.h>
#include<stdlib.h>
int min(int a, int b) {return a<b ? a : b; }
int min_of_max(int arr[], int n, int k) {
    int f=0,b = -1;
    int *deque = malloc(n*sizeof(int));
    int minMax = __INT_MAX__;
    
    for(int i=0;i<n;i++) {
        while(f<=b && deque[f] <= i - k) f++;
        while(f<=b && arr[deque[b]] <= arr[i]) b--;
        deque[++b]=i;
        if(i>=k -1) {
            minMax = min(minMax, arr[deque[f]]);
            
        }
    }
    free(deque);
    return minMax;
}
int main() {
    int n, q;
    scanf("%d %d", &n, &q);
    int arr[n];
    for(int i=0;i<n;i++) scanf("%d", &arr[i]);
    for(int i=0;i<q;i++) {
        int k;
        scanf("%d", &k);
        printf("%d\n", min_of_max(arr,n,k));
    }
    return 0;
}
