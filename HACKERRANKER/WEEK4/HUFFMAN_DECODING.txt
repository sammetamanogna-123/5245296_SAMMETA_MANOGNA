#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>

void decode(const char* s) {
    int i = 0;
    while (s[i] != '\0') {
        if(s[i] == '1') {
            printf("A");
            i++;
        } else if(s[i] == '0') {
            if(s[i+1] == '0') {
                printf("B");
                i += 2;
            } else if(s[i+1] == '1') {
                printf("C");
                i += 2;
            }
        }
    }
    printf("\n");
}
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    char s[] = "1001011";
    decode(s);
    return 0;
}
