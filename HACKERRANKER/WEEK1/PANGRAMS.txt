#include<stdio.h>

char* pangrams(char* s) {
    bool alpha_pre[26] = {false};
    int i;
    for (i = 0; s[i] != '\0'; i++) {
        char c = tolower(s[i]);
        if (c >= 'a' && c <= 'z') {
            alpha_pre[c - 'a'] = true;
        }
    }
    for (i = 0; i < 26; i++) {
        if (alpha_pre[i] == false) {
            return "not pangram";
        }
    }
    return "pangram";
}
