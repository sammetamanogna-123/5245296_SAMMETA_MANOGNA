#include<stdio.h>

int diagonalDifference(int arr_rows, int arr_columns, int** arr) {
    int l_to_r_s = 0;
    int r_to_l_s = 0;
    for (int i = 0; i < arr_rows; i++) {
        l_to_r_s += arr[i][i];
        r_to_l_s += arr[i][arr_columns - i - 1];
    }
    int dif = l_to_r_s - r_to_l_s;
    return (dif < 0) ? -dif : dif;
}
